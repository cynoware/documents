• 简介
    可以使用UART驱动LED客显，通常只要先设定好UART输出波特率， 然后按照LED的指令文档，输出显示指令就可以了。
    （以下代码仅做示例，针对不同的LED，波特率和指令系统可能不同。）
     
  • 示例代码
                String[] ledtitle = new String[]{"price","total","collect","givechange","Init"};

public void testLed(Device baseDevice){
    byte[] price = null;
    int uart_num = 0;
    	
    uart_num = config.CONFIG_ONBOARD_LED_UART; //config.CONFIG_ONBOARD_LED_UART = 6
    	
    if(ledtitle[titleindex].equalsIgnoreCase("price")){
    	price = new byte[]{0x1B,0x73,0x31};
    }
    else if(ledtitle[titleindex].equalsIgnoreCase("total")){
    	price = new byte[]{0x1B, 0x73, 0x32};
    }
    else if(ledtitle[titleindex].equalsIgnoreCase("collect")){
    	price = new byte[]{0x1B, 0x73, 0x33};
    }
    else if(ledtitle[titleindex].equalsIgnoreCase("givechange")){
    	price = new byte[]{0x1B, 0x73, 0x34};
    }
    else{//Init
    	price = new byte[]{0x1B, 0x40, 0x0C};
    }
    	
    UART.setConfig(baseDevice, uart_num, 2400, 8, UART.PRRITY_NONE, UART.STOPBITS_1);
    	 
    UART.fixedWrite(baseDevice, uart_num, new byte[]{0x1B, 0x40});
    if(led_sel != true){
    	msleep(500);
    }
    UART.fixedWrite(baseDevice, uart_num, price);
    if(led_sel != true){
        msleep(500);
    }
    if(showtxt.length() > 0){
        if(showtxt.length() < 8){
            int chars = 8-showtxt.length();
            for(int i = 0;i < chars;i++){
                showtxt=" " + showtxt;
            }
        }
        UART.fixedWrite(baseDevice, uart_num, new byte[]{0x1B, 0x51, 0x41});
            
        if(led_sel !=true){
            msleep(100);
        }
        UART.write(baseDevice, uart_num, showtxt.getBytes());
        UART.write(baseDevice, uart_num, new byte[]{0x0D});
    }
}

public boolean msleep(long time){
    try {
        Thread.sleep(time);
    } catch (InterruptedException e) {
        return false;
    }
    return true;
}
