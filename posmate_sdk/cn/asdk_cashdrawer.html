• 简介
    可以通过输出GPIO上的电平，实现打开钱箱的功能。具体步骤为，向钱箱的连线上输出高电平，等待200毫秒后，再输出低电平。
     
  • 示例代码
                   
//Example of cash drawer --
//This demo shows how to open cash draw by GPIO
private void mainCashDrawer(Device baseDevice){
    int CONFIG_CASHDRAWER_GPIO = (32*2+1);
    com.posmate.GPIO.setMode(baseDevice, CONFIG_CASHDRAWER_GPIO, 1, 0);
    	
    try {
        com.posmate.GPIO.output(baseDevice, CONFIG_CASHDRAWER_GPIO, 1);
        Thread.sleep(200);
        com.posmate.GPIO.output(baseDevice, CONFIG_CASHDRAWER_GPIO, 0);
        Thread.sleep(200);
    } catch (Exception ex) {
        ex.printStackTrace();
    }
}
